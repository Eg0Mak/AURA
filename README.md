# Структура проекта

rag_project/
├── data/
│   ├── raw/                 # исходные документы (PDF, DOCX, TXT и др.)
│   ├── processed/           # очищенные тексты документов
│   └── chunks/              # чанки документов с метаданными
│
├── embeddings/
│   ├── model_embed.py       # код загрузки модели эмбеддингов и вычисления векторов
│   └── vectors.faiss        # сохранённый индекс FAISS (или файл хранения векторов)
│
├── llm/
│   ├── model_gen.py         # код загрузки генераторной модели (LLM) и функции генерации
│   └── prompt_templates.py  # шаблоны промптов для генерации
│
├── retrieval/
│   ├── indexer.py           # код построения и обновления индекса векторного поиска
│   └── retriever.py         # код выполнения поиска по индексу и извлечения чанков
│
├── app/
│   ├── main.py              # основной скрипт/приложение (CLI или API)
│   └── api.py               # если будет REST-интерфейс или веб-сервис
│
├── tests/
│   ├── test_embeddings.py   # тесты для эмбеддинговой части
│   ├── test_retrieval.py    # тесты для retrieval части
│   └── test_generation.py   # тесты для генерации ответов
│
├── utils/
│   ├── file_utils.py        # вспомогательные функции работы с файлами/документами
│   ├── chunk_utils.py       # функции разбивки на чанки
│   └── config.py            # конфигурация проекта (пути, параметры)
│
├── .env                     # переменные окружения (если нужны)
├── requirements.txt         # список зависимостей проекта
└── README.md                # описание проекта, инструкция по запуску
